import { NextResponse } from 'next/server'; export async function POST(request: Request) { try { const { url } = await request.json(); if (!url) { return NextResponse.json({ success: false, message: 'لینک ارسال نشده است' }, { status: 400 }); } // ۱. تنظیم ورودی برای ارسال به Apify const input = { "tiktokUrls": [url], "apiVersion": "Musically", "proxySettings": { "useApifyProxy": true } }; // ۲. ارسال درخواست به تیک‌تاک دانلودر در Apify // به جای YOUR_APIFY_TOKEN باید توکن اکانت خودتان را بگذارید const response = await fetch('https://api.apify.com/v2/acts/wilcode~fast-tiktok-downloader-api/run-sync-get-dataset-items?token=YOUR_APIFY_TOKEN', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(input) }); const result = await response.json(); // ۳. بررسی موفقیت‌آمیز بودن و ارسال دیتای تیک‌تاک به فرانت‌اند سایت if (result && result[0] && result[0].success) { return NextResponse.json({ success: true, title: result[0].data.description, video: result[0].data.media.videoHD[0], audio: result[0].data.media.music, author: result[0].data.creator.name }); } return NextResponse.json({ success: false, message: 'خطا در دریافت اطلاعات از تیک‌تاک' }, { status: 500 }); } catch (error) { return NextResponse.json({ success: false, message: 'خطای سرور' }, { status: 500 }); } }
