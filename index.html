import { NextResponse } from 'next/server'; export async function POST(request: Request) { try { const { url } = await request.json(); if (!url) { return NextResponse.json({ success: false, message: 'لینک ارسال نشده است' }, { status: 400 }); } // ۱. ارسال لینک تیک‌تاک به API اختصاصی SnapTik برای پردازش const snapTikResponse = await fetch('https://api.snaptik.cx/api/download', { method: 'POST', headers: { 'Content-Type': 'application/json', 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36' }, body: JSON.stringify({ url: url }) }); const result = await snapTikResponse.json(); // ۲. استخراج لینک‌های دانلود مستقیم و بدون واترمارک از پاسخ SnapTik if (result && result.success && result.data) { return NextResponse.json({ success: true, title: result.data.title || 'TikTok Video', video: result.data.video_hd || result.data.video || result.data.links[0], // لینک مستقیم ویدیو بدون واترمارک audio: result.data.audio || result.data.music, // لینک موزیک author: result.data.author?.name || 'TikTok User' }); } return NextResponse.json({ success: false, message: 'SnapTik نتوانست ویدیو را پردازش کند' }, { status: 500 }); } catch (error) { return NextResponse.json({ success: false, message: 'خطای سرور در ارتباط با SnapTik' }, { status: 500 }); } }
